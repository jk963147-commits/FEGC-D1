                    <i class="ri-hammer-line text-4xl mb-2 block"></i>

                </div>

                <div v-for="(item, index) in engItems" :key="item.id" class="bg-white border border-gray-100 rounded-2xl p-5 ios-shadow space-y-4 animate-fade-in">
                    <div class="flex justify-between items-center pb-2 border-b border-gray-50">

    <script>
        const { createApp, ref, computed, watch } = Vue;

        createApp({
            setup() {
                const currentTab = ref('eng'); // 'eng' or 'doc'
                
                const engItems = ref([
                    { id: 1, type: '基樁', startDate: '', endDate: '', rebarDate: '', formworkDone: false, curingTime: '' }
                ]);
                
                const docItems = ref([]);

                // 邏輯：獲取開始日期
                const getStartDate = (index) => {
                    if (index === 0) {
                        return engItems.value[0].startDate;
                    } else {
                        // 自動連動上一項的結束日期
                        const prevDate = engItems.value[index - 1].endDate;
                        engItems.value[index].startDate = prevDate; // 同步數據
                        return prevDate;
                    }
                };

                const addItem = () => {
                    if (currentTab.value === 'eng') {
                        engItems.value.push({
                            id: Date.now(),
                            type: '基樁',
                            startDate: '', // 會被自動計算覆蓋
                            endDate: '',
                            rebarDate: '',
                            formworkDone: false,
                            curingTime: ''
                        });
                    } else {
                        docItems.value.push({
                            id: Date.now(),
                            type: '施工圖',
                            startDate: '',
                            endDate: ''
                        });
                    }
                };

                const removeEngItem = (index) => engItems.value.splice(index, 1);
                const removeDocItem = (index) => docItems.value.splice(index, 1);

                return {
                    currentTab,
                    engItems,
                    docItems,
                    addItem,
                    removeEngItem,
                    removeDocItem,
                    getStartDate
                };
            }
        }).mount('#app');
    </script>
